```yaml

license: MIT
```

Most skills omit `compatibility`; the one observed form is a free-form string. Schema shape (D-12) accommodates this — `compatibility` is optional and typed as `z.string()`. This keeps REQUIREMENTS §TEST-04 satisfied ("typed `compatibility`") without forcing a structured shape that doesn't exist in the current corpus:

```typescript
// scripts/lib/frontmatter.ts
import { z } from "zod";

/**
 * Required + optional fields for every morsels/skills/*/SKILL.md frontmatter.
 *
 * Per REQUIREMENTS §TEST-04:
 *   - name: string, equals the enclosing skill directory name (enforced in lint-skills.ts)
 *   - description: string, non-empty, ≤ 1024 chars
 *   - license: literal "MIT"
 *   - compatibility: typed (optional string — matches what the current corpus uses;
 *     no skill today ships a structured object, so a stricter shape would break
 *     the single skill that does set it: dragon-guard = "Designed for Pi (pi-coding-agent)")
 *
 * Unknown extra fields pass through (passthrough) — the schema is a minimum
 * contract, not a closed-world assertion. PITFALLS §4 ("schema half-life"):
 * do NOT loosen this schema in Phase 2; if a new field becomes a real
 * requirement, add it here and update morsels/AGENTS.md.
 */
export const SkillFrontmatterSchema = z
  .object({
    name: z.string().min(1),
    description: z.string().min(1).max(1024),
    license: z.literal("MIT"),
    compatibility: z.string().optional(),
  })
  .passthrough();

export type SkillFrontmatter = z.infer<typeof SkillFrontmatterSchema>;
```

If the executor discovers at author-time that a live skill uses a structured `compatibility` object (e.g. `{ models: [...], runtimes: [...] }`), the shape becomes a union — but only if actually in use. Otherwise stick with the string form. Do NOT pre-invent a structured shape; PITFALLS §4 warns against premature schema expansion.

### scripts/lint-skills.ts (new file — CLI walker)

```typescript
#!/usr/bin/env node
/**
 * morsels/skills/*/SKILL.md linter per REQUIREMENTS §TEST-04.
 *
 * Run via: node --experimental-strip-types scripts/lint-skills.ts
 * (also wired as `pnpm lint:skills` at repo root — see 02-01).
 *
 * Validates:
 *   - frontmatter against scripts/lib/frontmatter.ts SkillFrontmatterSchema
 *   - body text rejects stale Symbol.for("hoard.*") residue
 *   - body text rejects Symbol.for("pantry.<X>") where <X> is not a live
 *     PANTRY_KEYS key (ingested dynamically from berrygems/lib/globals.ts
 *     per D-14; regex fallback if dynamic import chokes)
 *
 * Exit codes per D-15:
 *   0 — clean run, all skills pass
 *   1 — any violation (frontmatter, body residue, or unknown pantry key)
 */
import { readFileSync, readdirSync, statSync } from "node:fs";
import { join, resolve, dirname } from "node:path";
import { fileURLToPath } from "node:url";
import { parse as parseYaml } from "yaml";
import { SkillFrontmatterSchema } from "./lib/frontmatter.ts";

const __dirname = dirname(fileURLToPath(import.meta.url));
const REPO_ROOT = resolve(__dirname, "..");
const MORSELS_ROOT = join(REPO_ROOT, "morsels", "skills");
const GLOBALS_PATH = join(REPO_ROOT, "berrygems", "lib", "globals.ts");

const FRONTMATTER_RE = /^---\r?\n([\s\S]*?)\r?\n---\r?\n([\s\S]*)$/;
const HOARD_SYMBOL_RE = /Symbol\.for\(\s*["']hoard\.([^"']+)["']\s*\)/g;
const PANTRY_SYMBOL_RE = /Symbol\.for\(\s*["']pantry\.([^"']+)["']\s*\)/g;

// ── Ingest live PANTRY_KEYS (D-14 preferred: dynamic import; fallback: regex) ──
async function loadPantryKeys(): Promise<Set<string>> {
  try {
    const mod = await import(GLOBALS_PATH);
    if (mod && typeof mod.PANTRY_KEYS === "object" && mod.PANTRY_KEYS !== null) {
      return new Set(Object.keys(mod.PANTRY_KEYS));
    }
    // Fall through to regex if the shape is unexpected.
  } catch {
    // Fall through to regex fallback on any import failure.
  }
  // Fallback: scrape keys from globals.ts source (D-14, case c).
  const src = readFileSync(GLOBALS_PATH, "utf-8");
  const match = src.match(/PANTRY_KEYS\s*=\s*\{([\s\S]*?)\}\s*as\s*const/);
  if (!match) {
    throw new Error(
      `lint-skills: unable to extract PANTRY_KEYS from ${GLOBALS_PATH}; update the fallback regex.`,
    );
  }
  const keys = new Set<string>();
  const keyRe = /^\s*([A-Za-z_][A-Za-z0-9_]*)\s*:/gm;
  let m: RegExpExecArray | null;
  while ((m = keyRe.exec(match[1]!)) !== null) {
    keys.add(m[1]!);
  }
  return keys;
}

// ── Main walk ──
async function main(): Promise<void> {
  const validPantryKeys = await loadPantryKeys();

  const skillDirs = readdirSync(MORSELS_ROOT).filter((d) =>
    statSync(join(MORSELS_ROOT, d)).isDirectory(),
  );

  let totalSkills = 0;
  let totalViolations = 0;
  const failedSkills: string[] = [];

  for (const dir of skillDirs.sort()) {
    const skillPath = join(MORSELS_ROOT, dir, "SKILL.md");
    let raw: string;
    try {
      raw = readFileSync(skillPath, "utf-8");
    } catch {
      // Skip directories without a SKILL.md (shouldn't happen, but be safe).
      continue;
    }
    totalSkills++;

    const issues: string[] = [];

    // (1) Frontmatter block presence.
    const fmMatch = raw.match(FRONTMATTER_RE);
    if (!fmMatch) {
      issues.push(`missing '---\\n...\\n---\\n' frontmatter block at top of file`);
    } else {
      // (2) YAML parse.
      let parsed: unknown;
      try {
        parsed = parseYaml(fmMatch[1]!);
      } catch (err) {
        issues.push(`YAML parse error: ${(err as Error).message}`);
      }

      // (3) Zod validate.
      if (parsed !== undefined) {
        const result = SkillFrontmatterSchema.safeParse(parsed);
        if (!result.success) {
          for (const issue of result.error.issues) {
            issues.push(
              `frontmatter.${issue.path.join(".") || "(root)"}: ${issue.message}`,
            );
          }
        } else {
          // Extra cross-check per REQUIREMENTS §TEST-04: name === directory.
          if (result.data.name !== dir) {
            issues.push(
              `frontmatter.name "${result.data.name}" does not match skill directory "${dir}"`,
            );
          }
        }
      }

      // (4) Body scans — always run if body is present, regardless of frontmatter validity.
      const body = fmMatch[2] ?? "";
      HOARD_SYMBOL_RE.lastIndex = 0;
      let hm: RegExpExecArray | null;
      while ((hm = HOARD_SYMBOL_RE.exec(body)) !== null) {
        issues.push(
          `body contains stale Symbol.for("hoard.${hm[1]}") residue (pantry amputation complete — remove)`,
        );
      }
      PANTRY_SYMBOL_RE.lastIndex = 0;
      let pm: RegExpExecArray | null;
      while ((pm = PANTRY_SYMBOL_RE.exec(body)) !== null) {
        const key = pm[1]!;
        if (!validPantryKeys.has(key)) {
          issues.push(
            `body references unknown Symbol.for("pantry.${key}") — not a key of PANTRY_KEYS (known: ${[...validPantryKeys].sort().join(", ")})`,
          );
        }
      }
    }

    if (issues.length > 0) {
      totalViolations += issues.length;
      failedSkills.push(dir);
      // Per-skill grouped diagnostics block per D-15.
      console.error(`\n✗ ${dir}:`);
      for (const issue of issues) {
        console.error(`    - ${issue}`);
      }
    }
  }

  if (totalViolations > 0) {
    console.error(
      `\n✗ lint-skills: ${failedSkills.length} of ${totalSkills} skills failed (${totalViolations} violations): ${failedSkills.join(", ")}`,
    );
    process.exit(1);
  } else {
    console.log(`✓ lint-skills: all ${totalSkills} skills passed`);
    process.exit(0);
  }
}

main().catch((err) => {
  console.error("lint-skills crashed:", err);
  process.exit(2);
});
```

**Critical design notes:**

- The dynamic import path in `loadPantryKeys()` tries `await import(GLOBALS_PATH)` — which under `node --experimental-strip-types` resolves the `.ts` file directly. If Node's strip-types implementation cannot follow the path (e.g. ESM resolution quirks with absolute paths), the `try/catch` drops into the regex fallback, which is the D-14 (c) path. Both paths yield a `Set<string>` of keys.
- The regex fallback is a dead branch in practice on the current Node version (Phase 1's globals.ts is a clean export), but it's the documented escape hatch. When it IS hit, the executor should open a follow-up todo to re-enable the dynamic path against the newer Node version.
- `FRONTMATTER_RE` tolerates `\r\n` for Windows-line-ending files (some editors emit these; Phase 1 files are LF but future contributions may not be).
- Body-scan regexes use `lastIndex = 0` before each loop because they're module-scope `g` flag regexes — stateful between files without a reset.
- The `name !== directory` check is enforced per REQUIREMENTS §TEST-04 wording ("name: string, equals directory name").
- No `console.log` beyond the final success line — all diagnostic output goes to `stderr` via `console.error` so a caller can separate success/failure streams.
