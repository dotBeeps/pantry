```typescript
/**
 * Install smoke test — harness-fast side (Phase 2 half of the dual-smoke gate).
 *
 * Calls verifySandboxInstall({ packageDir }) from @marcfargas/pi-test-harness
 * against the repo root, then asserts NAMED extensions + skills per PITFALLS §5:
 * a rename silently passes a count-only check but fails a name check loudly —
 * which is the regression pressure we want.
 *
 * Excluded from the default vitest run via berrygems/vitest.config.ts
 * (test.include omits tests/smoke/** per 02-CONTEXT.md §D-17). Reachable via
 * `pnpm --dir berrygems test:smoke` or Phase 4 CI-02's same invocation.
 *
 * Phase 4 CI-02 consumes this file verbatim as the harness-fast gate and adds
 * a separate shell step running `HOME=$(mktemp -d) pi install $GITHUB_WORKSPACE
 * && pi list` as the real-install gate. Phase 2 is the author; Phase 4 is the
 * wirer.
 */
import { describe, it, expect } from "vitest";
import { verifySandboxInstall } from "@marcfargas/pi-test-harness";
import { resolve, dirname } from "node:path";
import { fileURLToPath } from "node:url";

const __dirname = dirname(fileURLToPath(import.meta.url));
const REPO_ROOT = resolve(__dirname, "..", "..", ".."); // berrygems/tests/smoke → repo root

describe("install smoke (harness-fast — verifySandboxInstall)", () => {
  it("installs the package and loads named extensions + skills", async () => {
    const result = await verifySandboxInstall({ packageDir: REPO_ROOT });

    // Primary: NAMED assertions (PITFALLS §5 — named > count).
    const extensionNames = (result.extensions ?? []).map(
      (e: { name?: string }) => e?.name ?? "",
    );
    expect(extensionNames).toEqual(
      expect.arrayContaining(["dragon-parchment", "dragon-guard"]),
    );

    const skillNames = (result.skills ?? []).map(
      (s: { name?: string }) => s?.name ?? "",
    );
    expect(skillNames).toEqual(expect.arrayContaining(["git"]));

    // Secondary: count sanity. If counts drop sharply (e.g. extensions went from 17 → 3
    // because a loader regression silently dropped them), this is a second trip wire.
    expect(result.extensions?.length ?? 0).toBeGreaterThanOrEqual(17);
    expect(result.skills?.length ?? 0).toBeGreaterThanOrEqual(54);
  }, 120_000); // install is slow (npm pack + install); 2 min timeout
});
```

### Key design constraints

- **`packageDir` resolution.** `__dirname` is the smoke directory; two `..` take us to `berrygems/tests/`, a third to `berrygems/`, a fourth to the repo root. Per RESEARCH §9 the target is repo root (where the root `package.json` with the `pi.extensions` / `pi.skills` manifest lives). Count the `..`s in the final file against the actual directory depth; the right value is `resolve(__dirname, "..", "..", "..")` if the file is at `berrygems/tests/smoke/install.test.ts`.
- **Named assertions** must reference extensions/skills known to load — dragon-parchment and dragon-guard are the target pair because 02-04 ships their tests first; `git` is a stable, well-known morsel in the corpus. Do NOT reference speculative names (e.g. "dragon-whatever") — the whole point of the name check is that the name is real.
- **Timeout 120s.** `npm pack` + install on a warm cache typically takes 10-30s; cold cache can hit 60-90s. 120s is a defensive ceiling.
- **Return-shape handling.** The test uses defensive accessors (`result.extensions ?? []`, `e?.name ?? ""`) because the harness's precise shape for `verifySandboxInstall` is not fully documented at plan-authoring time. 02-03's spike may have surfaced the real shape — reuse whatever `session.*` vs `result.*` pattern 02-03-SUMMARY.md documented. If the shape is crisper than the defensive pattern suggests, tighten the accessors.
- **No `session.close()`.** `verifySandboxInstall` is one-shot (performs an install, returns a result summary) — unlike `createTestSession`, it doesn't return a long-lived session that needs closing.
